===== .github/workflows/build-and-push.yml =====
name: Build and Push Images

on:
  push:
    branches: [main]
  workflow_dispatch: {}

env:
  AWS_REGION: ap-southeast-1

permissions:
  id-token: write
  contents: read

jobs:
  build-push:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        include:
          - service: agent-worker
            cmd_path: ./cmd/worker
          - service: dispatcher
            cmd_path: ./cmd/dispatcher
          - service: reaper
            cmd_path: ./cmd/heartbeat-reaper
          - service: salience-decay
            cmd_path: ./cmd/salience-decay
          - service: scoring-api
            cmd_path: ./cmd/scoring-api
          - service: review-api
            cmd_path: ./cmd/review-api

    steps:
      - uses: actions/checkout@v4

      - name: Configure AWS credentials via OIDC
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: ${{ secrets.AWS_GITHUB_ACTIONS_ROLE_ARN }}
          aws-region: ${{ env.AWS_REGION }}

      - name: Login to ECR
        id: ecr-login
        uses: aws-actions/amazon-ecr-login@v2

      - name: Build and push ${{ matrix.service }}
        env:
          ECR_REGISTRY: ${{ steps.ecr-login.outputs.registry }}
          IMAGE_TAG: ${{ github.sha }}
        run: |
          REPO="${ECR_REGISTRY}/hivemind/dev/${{ matrix.service }}"
          docker build -f Dockerfile.lambda-go \
            --build-arg SERVICE_PATH=${{ matrix.cmd_path }} \
            -t "${REPO}:${IMAGE_TAG}" \
            -t "${REPO}:latest" .
          docker push "${REPO}:${IMAGE_TAG}"
          docker push "${REPO}:latest"

  build-push-scoring-python:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Configure AWS credentials via OIDC
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: ${{ secrets.AWS_GITHUB_ACTIONS_ROLE_ARN }}
          aws-region: ${{ env.AWS_REGION }}

      - name: Login to ECR
        id: ecr-login
        uses: aws-actions/amazon-ecr-login@v2

      - name: Build and push scoring-python
        env:
          ECR_REGISTRY: ${{ steps.ecr-login.outputs.registry }}
          IMAGE_TAG: ${{ github.sha }}
        run: |
          REPO="${ECR_REGISTRY}/hivemind/dev/scoring-python"
          docker build -f services/scoring-python/Dockerfile \
            -t "${REPO}:${IMAGE_TAG}" \
            -t "${REPO}:latest" .
          docker push "${REPO}:${IMAGE_TAG}"
          docker push "${REPO}:latest"

===== .github/workflows/ci.yml =====
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  go-build-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-go@v5
        with:
          go-version: '1.25'

      - name: Build
        run: go build ./...

      - name: Unit tests
        run: go test ./... -v

  go-fmt-check:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-go@v5
        with:
          go-version: '1.25'
      - name: Check gofmt
        run: |
          UNFORMATTED=$(gofmt -l .)
          if [ -n "$UNFORMATTED" ]; then
            echo "Files not gofmt'd:"
            echo "$UNFORMATTED"
            exit 1
          fi

  go-vet:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-go@v5
        with:
          go-version: '1.25'
      - name: Vet
        run: go vet ./...

  python-scoring-service:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: '3.12'
      - name: Install dependencies
        run: pip install --no-cache-dir -r services/scoring-python/requirements.txt
      - name: Syntax check
        run: python -m py_compile services/scoring-python/main.py
      - name: Import check
        run: python -c "import sys; sys.path.insert(0, 'services/scoring-python'); import main"

  docker-build-check:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        service: [worker, dispatcher, heartbeat-reaper, salience-decay, scoring-api, review-api]
    steps:
      - uses: actions/checkout@v4
      - name: Build ${{ matrix.service }} image
        run: |
          docker build -f Dockerfile.lambda-go \
            --build-arg SERVICE_PATH=./cmd/${{ matrix.service }} \
            -t hivemind-${{ matrix.service }}:ci .

  docker-build-scoring-python:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Build scoring-python image
        run: |
          docker build -f services/scoring-python/Dockerfile \
            -t hivemind-scoring-python:ci .

  terraform-validate:
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: terraform
    steps:
      - uses: actions/checkout@v4
      - uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: '1.9.0'
      - name: Init (backend only, no state access needed for validate)
        run: terraform init -backend=false
      - name: Validate
        run: terraform validate
      - name: Format check
        run: terraform fmt -check -recursive

===== .github/workflows/deploy-canary.yml =====
name: Deploy Canary Production

on:
  workflow_run:
    workflows: ["Deploy Staging"]
    types: [completed]
    branches: [main]

env:
  AWS_REGION: ap-southeast-1
  PROJECT: hivemind
  ENVIRONMENT: dev
  CANARY_WEIGHT: 0.10
  CANARY_WAIT_SECONDS: 300

permissions:
  id-token: write
  contents: read

jobs:
  canary-deploy:
    if: ${{ github.event.workflow_run.conclusion == 'success' }}
    runs-on: ubuntu-latest
    strategy:
      matrix:
        service: [agent-worker, scoring-api, scoring-python, review-api]
      fail-fast: false

    steps:
      - name: Configure AWS credentials via OIDC
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: ${{ secrets.AWS_GITHUB_ACTIONS_ROLE_ARN }}
          aws-region: ${{ env.AWS_REGION }}

      - name: Get function and current version info
        id: info
        run: |
          FUNCTION_NAME="${PROJECT}-${ENVIRONMENT}-${{ matrix.service }}"
          NEW_VERSION=$(aws lambda list-versions-by-function \
            --function-name "$FUNCTION_NAME" \
            --query 'max_by(Versions, &to_number(Version))' \
            --output json | jq -r '.Version')
          CURRENT_ALIAS=$(aws lambda get-alias \
            --function-name "$FUNCTION_NAME" \
            --name live)
          OLD_VERSION=$(echo "$CURRENT_ALIAS" | jq -r '.FunctionVersion')

          echo "function_name=$FUNCTION_NAME" >> "$GITHUB_OUTPUT"
          echo "new_version=$NEW_VERSION" >> "$GITHUB_OUTPUT"
          echo "old_version=$OLD_VERSION" >> "$GITHUB_OUTPUT"

      - name: Shift 10% traffic to new version
        run: |
          aws lambda update-alias \
            --function-name "${{ steps.info.outputs.function_name }}" \
            --name live \
            --function-version "${{ steps.info.outputs.old_version }}" \
            --routing-config "AdditionalVersionWeights={${{ steps.info.outputs.new_version }}=${CANARY_WEIGHT}}"

          echo "Canary started: ${CANARY_WEIGHT} traffic -> version ${{ steps.info.outputs.new_version }}"

      - name: Wait and observe
        run: sleep ${CANARY_WAIT_SECONDS}

      - name: Check CloudWatch error alarm
        id: check
        run: |
          ALARM_NAME="${PROJECT}-${ENVIRONMENT}-${{ matrix.service }}-errors"
          STATE=$(aws cloudwatch describe-alarms \
            --alarm-names "$ALARM_NAME" \
            --query 'MetricAlarms[0].StateValue' \
            --output text)
          echo "alarm_state=$STATE" >> "$GITHUB_OUTPUT"
          echo "Alarm ${ALARM_NAME} state: ${STATE}"

      - name: Promote to 100% (alarm OK)
        if: steps.check.outputs.alarm_state != 'ALARM'
        run: |
          aws lambda update-alias \
            --function-name "${{ steps.info.outputs.function_name }}" \
            --name live \
            --function-version "${{ steps.info.outputs.new_version }}" \
            --routing-config "{}"
          echo "Promoted: 100% traffic -> version ${{ steps.info.outputs.new_version }}"

      - name: Rollback (alarm triggered)
        if: steps.check.outputs.alarm_state == 'ALARM'
        run: |
          aws lambda update-alias \
            --function-name "${{ steps.info.outputs.function_name }}" \
            --name live \
            --function-version "${{ steps.info.outputs.old_version }}" \
            --routing-config "{}"
          echo "Rolled back: 100% traffic -> version ${{ steps.info.outputs.old_version }} (canary failed)"
          exit 1

  deploy-scheduled-services:
    if: ${{ github.event.workflow_run.conclusion == 'success' }}
    runs-on: ubuntu-latest
    strategy:
      matrix:
        service: [dispatcher, reaper, salience-decay]

    steps:
      - name: Configure AWS credentials via OIDC
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: ${{ secrets.AWS_GITHUB_ACTIONS_ROLE_ARN }}
          aws-region: ${{ env.AWS_REGION }}

      - name: Promote alias directly (low-risk scheduled service)
        run: |
          FUNCTION_NAME="${PROJECT}-${ENVIRONMENT}-${{ matrix.service }}"
          NEW_VERSION=$(aws lambda list-versions-by-function \
            --function-name "$FUNCTION_NAME" \
            --query 'max_by(Versions, &to_number(Version))' \
            --output json | jq -r '.Version')

          aws lambda update-alias \
            --function-name "$FUNCTION_NAME" \
            --name live \
            --function-version "$NEW_VERSION" \
            --routing-config "{}"

          echo "Deployed ${{ matrix.service }} -> version ${NEW_VERSION} (no canary, scheduled service)"

===== .github/workflows/deploy-staging.yml =====
name: Deploy Staging

on:
  workflow_run:
    workflows: ["Build and Push Images"]
    types: [completed]
    branches: [main]

env:
  AWS_REGION: ap-southeast-1

permissions:
  id-token: write
  contents: read

jobs:
  deploy-staging:
    if: ${{ github.event.workflow_run.conclusion == 'success' }}
    runs-on: ubuntu-latest
    outputs:
      image_tag: ${{ steps.set-tag.outputs.image_tag }}
    steps:
      - uses: actions/checkout@v4

      - name: Configure AWS credentials via OIDC
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: ${{ secrets.AWS_GITHUB_ACTIONS_ROLE_ARN }}
          aws-region: ${{ env.AWS_REGION }}

      - name: Set image tag from triggering commit
        id: set-tag
        run: echo "image_tag=${{ github.event.workflow_run.head_sha }}" >> "$GITHUB_OUTPUT"

      - uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: '1.9.0'

      - name: Terraform Init
        working-directory: terraform
        run: terraform init

      - name: Terraform Apply (update image_tag var)
        working-directory: terraform
        env:
          TF_VAR_cockroachdb_connection_string: ${{ secrets.DATABASE_URL }}
          TF_VAR_cockroachdb_mcp_endpoint: ${{ secrets.COCKROACHDB_MCP_ENDPOINT }}
          TF_VAR_alert_email: ${{ secrets.ALERT_EMAIL }}
          TF_VAR_image_tag: ${{ steps.set-tag.outputs.image_tag }}
        run: terraform apply -auto-approve

  integration-test:
    needs: deploy-staging
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-go@v5
        with:
          go-version: '1.25'

      - name: Run integration tests against real CockroachDB
        env:
          DATABASE_URL: ${{ secrets.DATABASE_URL }}
        run: go test ./internal/memory/... -tags=integration -v

===== .github/workflows/terraform-infra.yml =====
name: Terraform Infrastructure

on:
  pull_request:
    paths:
      - 'terraform/**'
  workflow_dispatch:
    inputs:
      action:
        description: 'Terraform action'
        required: true
        default: 'plan'
        type: choice
        options:
          - plan
          - apply

permissions:
  id-token: write
  contents: read
  pull-requests: write

jobs:
  terraform:
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: terraform
    steps:
      - uses: actions/checkout@v4

      - uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: '1.9.0'

      - name: Configure AWS credentials via OIDC
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: ${{ secrets.AWS_GITHUB_ACTIONS_ROLE_ARN }}
          aws-region: ap-southeast-1

      - name: Terraform Init
        run: terraform init

      - name: Terraform Plan
        id: plan
        env:
          TF_VAR_cockroachdb_connection_string: ${{ secrets.DATABASE_URL }}
          TF_VAR_cockroachdb_mcp_endpoint: ${{ secrets.COCKROACHDB_MCP_ENDPOINT }}
          TF_VAR_alert_email: ${{ secrets.ALERT_EMAIL }}
        run: terraform plan -no-color -out=tfplan
        continue-on-error: true

      - name: Comment Plan on PR
        if: github.event_name == 'pull_request'
        uses: actions/github-script@v7
        with:
          script: |
            const output = `#### Terraform Plan
            \`\`\`
            ${{ steps.plan.outputs.stdout }}
            \`\`\`
            `;
            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: output
            });

      - name: Terraform Apply
        if: github.event_name == 'workflow_dispatch' && github.event.inputs.action == 'apply'
        env:
          TF_VAR_cockroachdb_connection_string: ${{ secrets.DATABASE_URL }}
          TF_VAR_cockroachdb_mcp_endpoint: ${{ secrets.COCKROACHDB_MCP_ENDPOINT }}
          TF_VAR_alert_email: ${{ secrets.ALERT_EMAIL }}
        run: terraform apply -auto-approve tfplan
